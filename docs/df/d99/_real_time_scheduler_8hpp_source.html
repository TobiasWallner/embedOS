240 void promote(){
241 // promote await back into running queue
242 for(RTTask* task : this->_await_bench){
243 if(!task->is_awaiting()){
244 logger::log_move(Clock::now(), task->name(), task->id(), "await", "run");
245 this->running_queue().push(task);
246 }
247 }
248 this->_await_bench.erase_if([](const RTTask* task){return !task->is_awaiting();});
249
250 // promote waiting queue into running queue
251 while(!this->waiting_queue().empty()){
252 RTTask* task = this->waiting_queue().top();
253 const time_point now = Clock::now();
254 if(task->ready_time() <= now){
255 logger::log_move(now, task->name(), task->id(), "wait", "run");
256 this->waiting_queue().pop();
257 this->running_queue().push(task);
258 }else{
259 return;
260 }
261 }
262 }
263
278 void run_next(){
279 if(this->running_queue().empty()){
280 FSleepUntil(this->waiting_queue().top()->ready_time());
281 }else{
282 RTTask* task = this->running_queue().top_pop();
283 task->_execution_start = Clock::now();
284 task->resume();
285 logger::log_resume(task->_execution_start, Clock::now(), task->name(), task->id());
286 // re-schedule
287
288 const CoTaskSignal signal = task->get_signal();
289 switch(signal.type()){
290 case CoTaskSignal::Type::Await : {
291 // clear signal
292 this->_await_bench.emplace_back(task);
293 logger::log_move(Clock::now(), task->name(), task->id(), "resume", "await");
294 }break;
295 case CoTaskSignal::Type::NextCycle : {
296 task->_schedule = task->next_schedule(task->_schedule, ExecutionTime<Clock>{task->_execution_start, Clock::now()});
297 this->waiting_queue().push(task);
298 logger::log_move(Clock::now(), task->name(), task->id(), "resume", "wait");
299 }break;
300 case CoTaskSignal::Type::ImplicitDelay : {
301 const duration rel_deadline = task->_schedule.deadline - task->_schedule.ready;
302 task->_schedule.ready = Clock::now() + std::chrono::duration_cast<duration>(signal.implicit_delay().delay);
303 task->_schedule.deadline = task->_schedule.ready + rel_deadline;
304 this->waiting_queue().push(task);
305 logger::log_move(Clock::now(), task->name(), task->id(), "resume", "wait");
306 }break;
307 case CoTaskSignal::Type::ExplicitDelay : {
308 task->_schedule.ready = Clock::now() + std::chrono::duration_cast<duration>(signal.explicit_delay().delay);
309 task->_schedule.deadline = task->_schedule.ready + std::chrono::duration_cast<duration>(signal.explicit_delay().rel_deadline);
310 logger::log_move(Clock::now(), task->name(), task->id(), "resume", "wait");
311 }break;
312 case CoTaskSignal::Type::None : {
313 /*
314 - Nothing to do
315 - task probably finished
316 - let it die by not inserting it into lists
317 */
318 logger::log_delete(Clock::now(), task->name(), task->id());
319 } break;
320 default : {
321 /*
322 - Nothing to do
323 - task probably finished
324 - let it die by not inserting it into lists
325 */
326 logger::log_delete(Clock::now(), task->name(), task->id());
327 } break;
328 }
329 }
